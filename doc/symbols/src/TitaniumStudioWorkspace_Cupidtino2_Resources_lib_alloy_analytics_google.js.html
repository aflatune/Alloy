1 include('../extern/analytics.js');
  2 Alloy.analytics = new JS.Singleton(Alloy.AnalyticsBase, {
  3   initialize: function() {
  4     this.analytics = new Analytics(AlloyConfig.analytics.key);
  5     this.callSuper();
  6   },
  7   
  8   start: function(dispatchPeriod){
  9     this.callSuper();
 10     this.analytics.start(dispatchPeriod);
 11   }, 
 12   
 13   stop: function(){
 14     this.callSuper();
 15     this.analytics.stop();
 16   }, 
 17   
 18   trackPageview: function(pageUrl){
 19     this.callSuper();
 20     this.analytics.trackPageview(pageUrl);
 21   },
 22   
 23   trackEvent: function(category, action, label, value){
 24     this.callSuper();
 25 
 26     // Use page views instead of events till issue is fixed - https://github.com/rogchap/Titanium-Google-Analytics/issues/6
 27     //if (this.analytics) {
 28     //  this.analytics.trackEvent(category, action, label, value);
 29     //}
 30     
 31     var url = [];
 32     url.push('/event/');
 33     url.push(category.replace(/\:/g, '_'));
 34     url.push('/');
 35     url.push(action);
 36     
 37     var params = [];
 38     if (label) {
 39       params.push('label=');
 40       params.push(encodeURIComponent(label));
 41     }
 42     
 43     if (value) {
 44       if (params.length > 0)
 45         params.push('&');
 46       params.push('value=');
 47       params.push(encodeURIComponent(value));
 48     }
 49     
 50     if (params.length > 0) {
 51       url.push('?');
 52       url.push(params.join(''));
 53     }
 54     
 55     url = url.join('');
 56     this.trackPageview(url);
 57   },
 58   
 59   reset:function(){
 60     this.callSuper();
 61     this.analytics.reset();
 62   }
 63 })
