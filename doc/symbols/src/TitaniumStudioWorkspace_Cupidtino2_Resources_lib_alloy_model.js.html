1 // The Superclass for all models
  2 // loosely basesd on ActiveRecord
  3 
  4 Alloy.Model = new JS.Class({
  5   _fields: {},
  6   attributes: {},
  7   new_record: true,
  8   db: null,
  9 
 10   // initialize the model. Needs to have the database passed
 11   // var item = new Item(db);
 12   //
 13   // the attributes hash can be used to pre-define attributes of the object:
 14   //
 15   // var item = new Item(db, {name: "Joe Doe"};
 16   //
 17   init: function(aDb, attributes) {
 18     this.db = aDb;
 19     if (attributes === undefined) {
 20       attributes = {};
 21     }
 22     _(this._fields).chain().each(function(value, key) {
 23 
 24       try {
 25         if (attributes[key] === undefined) {
 26           if (value === String) {
 27             this[key] = "";
 28           } else if (value === Date) {
 29             this[key] = new Date();
 30           } else {
 31             this[key] = 0;
 32           }
 33         } else
 34         {
 35 
 36           this[key] = attributes[key];
 37         }
 38 
 39       } catch(exception) {
 40         Ti.API.error(exception);
 41       }
 42     }, this);
 43     if (attributes.id === undefined) {
 44       this.new_record = true;
 45     }
 46     else
 47     {
 48       this.id = attributes.id;
 49       this.new_record = false;
 50     }
 51     //Ti.API.debug(this);
 52 
 53     this.attributes = attributes;
 54   },
 55 
 56   field_names: function() {
 57     _(this._fields).keys();
 58   },
 59 
 60   // find all objects
 61   // understands the following options:
 62   // {order: "name ASC"} -> use the parameter in the ORDER BY clause
 63   // {conditions: "bla = 123"} -> use the parameter to limit the query
 64   // {limit: 5} -> only return 5 elements
 65   //
 66   // item.find_all({order: "name ASC", conditions: "name like '%abc%'", limit: 10})
 67   find_all: function(options) {
 68     if ( !options ) options = {};
 69 
 70     var order = options.order;
 71     delete(options['order']);
 72     var condition = options.conditions;
 73     delete(options['conditions']);
 74     var limit = options.limit;
 75     delete(options['limit']);
 76     var order_sql = "";
 77     if (order) {
 78       order_sql = " ORDER BY " + order + " ";
 79     }
 80     var conditions_sql = "";
 81     if (condition) {
 82       conditions_sql = " WHERE " + condition + " ";
 83     }
 84     var limit_sql = "";
 85     if (limit) {
 86       limit_sql = " LIMIT " + limit + " ";
 87     }
 88 
 89     var sql = 'SELECT * from ' + this.table_name + conditions_sql + order_sql + limit_sql + ";";
 90     Titanium.API.debug(sql);
 91     var rows = this.db.execute(sql);
 92     var data = [];
 93     while (rows.isValidRow())
 94     {
 95       data.push(this.item_from(rows));
 96       rows.next();
 97     }
 98     rows.close();
 99     return data;
100   },
101 
102 
103   // needs to be "ove
104   _find: function(id) {
105     try {
106     var sql = 'SELECT * from ' + this.table_name + ' WHERE id = ?;';
107     var rows = this.db.execute(sql, id);
108     if (rows.getRowCount() > 0 && rows.isValidRow()) {
109       var item = this.item_from(rows);
110 
111     }
112     rows.close();
113     }
114     catch(exception) {
115       Ti.API.error(exception);
116       return null;
117     }
118     return item;
119 
120   },
121 
122   // store a new object in the db
123   create: function() {
124     var fields_sql = " (" + _(this._fields).keys().filter(function(key) {
125         return key != "id";
126     }, this).join(', ') + ") ";
127 
128     var placeholders = "(" + _(this._fields).chain().keys().filter(function(key) {
129         return key != "id";
130     }, this).map(function(field) {
131       return this.to_sql(field);
132     }, this).value().join(", ") + ") ";
133 
134     var rows = this.db.execute('INSERT INTO ' + this.table_name + fields_sql + ' VALUES ' + placeholders);
135     this.id = db.lastInsertRowId;
136   },
137 
138 
139   // update an existing object
140   update: function() {
141     var placeholders = _(this._fields).chain().keys().map(function(field) {
142       return field + " = " + this.to_sql(field);
143     }, this).value().join(", ");
144 
145     var rows = this.db.execute('UPDATE ' + this.table_name + ' SET ' + placeholders + ' WHERE id = ' + this.id);
146   },
147 
148   // saves the current object. Distinguishes between a new object (and then calls create)
149   // or an existing object
150   save: function() {
151     if (this.new_record) {
152       this.create();
153     } else
154     {
155       this.update();
156     }
157     this.new_record = false;
158     return true;
159   },
160 
161   // delete the object from the database
162   // usage:
163   // item = item.find(123);
164   // item.destroy();
165   //
166   // or
167   //
168   // item.destroy(123);
169   //
170 
171   destroy: function(id) {
172     var destroy_id;
173     if (id === undefined) {
174       destroy_id = this.id;
175     } else
176     {
177       destroy_id = id;
178     }
179     this.db.execute('DELETE FROM ' + this.table_name + ' WHERE id = ?', destroy_id);
180   },
181 
182 
183 
184   // this needs to be "redefined" in the actual class as "item_from"
185   // to create the correct type of Class. See examples
186   _item_from: function(row) {
187     var item = {};
188     _(this._fields).chain().keys().each(function(field) {
189       switch (this._fields[field]) {
190         case String:
191           item[field] = row.fieldByName(field);
192           break;
193 
194         case Number:
195           item[field] = row.fieldByName(field);
196           break;
197 
198         case Date:
199           item[field] = new Date(row.fieldByName(field));
200           break;
201 
202       }
203     }, this).value();
204     item.id = row.field(0);
205     return item;
206   },
207 
208   // turn the content of a field into an SQL representation
209   to_sql: function(field) {
210     switch (this._fields[field]) {
211       case String:
212         return "'" + this[field].toString() + "'";
213 
214       case Number:
215         return parseInt(this[field])
216 
217       case Date:
218         return "'" + this[field].toString() + "'";
219 
220     }
221   }
222 
223 });
