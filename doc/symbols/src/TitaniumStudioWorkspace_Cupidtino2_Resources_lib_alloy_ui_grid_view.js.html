 36 Alloy.UI.GridView = new JS.Class({
 37   initialize: function(params) {
 38     params = params || {};
 39     this.itemWidth = params.itemWidth || 50;
 40     this.itemHeight = params.itemHeight || 50;
 41     this.minMargin = params.minMargin || 0;
 42     this.view = new TableView({
 43       focusable: false,
 44       separatorStyle: Titanium.UI.iPhone.TableViewSeparatorStyle.NONE,
 45       allowsSelection: false
 46     });
 47   },
 48   
 49   render: function(params) {
 50     
 51   },
 52   
 53   _getTotalWidth: function() {
 54     if (Ti.UI.currentWindow)
 55       width = Ti.UI.currentWindow.width;
 56     else if (App && App.tabGroup)
 57       width = App.tabGroup.width;
 58     else
 59       width = Ti.Platform.displayCaps.platformWidth;
 60       
 61     return width;
 62   },
 63   
 64   setData: function(gridViewItems) {
 65     this.gridViewItems = gridViewItems;
 66     this.refresh();
 67   },
 68   
 69   refresh: function() {
 70     if (this._refreshing)
 71       return;
 72 
 73     this._refreshing = true;
 74     this.tableData = [];
 75     var currentGridViewRow = null;
 76 
 77     var totalWidth = this._getTotalWidth();
 78     var itemsPerRow = Math.floor((totalWidth - this.minMargin) / (this.itemWidth + this.minMargin));
 79     var margin = (totalWidth - (this.itemWidth * itemsPerRow)) / (itemsPerRow + 1);
 80     
 81     var itemsInCurrentRow = 0;
 82     var currentHeader = "~~NOHEADER~~";
 83     
 84     for (var i = 0; i < this.gridViewItems.length; i++) {
 85       var item = this.gridViewItems[i];
 86       
 87       var newHeader = false;
 88       newHeader = (currentHeader != item.header);
 89       if (newHeader)
 90         currentHeader = item.header;
 91       
 92       if (newHeader || itemsInCurrentRow == 0) {
 93         itemsInCurrentRow = 0;
 94         currentGridViewRow = new Alloy.UI.GridViewRow();
 95 
 96         if (newHeader) {
 97           this.startNewSection(item.header);
 98         }
 99         
100         this.tableData[this.tableData.length-1].add(currentGridViewRow.view);
101         currentGridViewRow.rowIndex = this.tableData[this.tableData.length-1].length - 1;
102       }
103             
104       item.view.left = margin + ((this.itemWidth + margin) * itemsInCurrentRow);
105       currentGridViewRow.add(item);
106       itemsInCurrentRow = (itemsInCurrentRow + 1) % itemsPerRow;
107     }
108     
109     this.view.setData(this.tableData);    
110     this._refreshing = false;
111   },
112    
113   startNewSection: function(header) {
114     var section = Ti.UI.createTableViewSection();
115     if (header) {
116       if (typeof(header) == "string") {
117         var v = new View({height: 30, backgroundColor: '#e0e0e0', opacity: 0.9});
118         var l = new Label({text: header, width: '100%', height: 30, textAlign: 'center', font: {fontWeight: 'bold', fontSize: 12}});
119         v.add(l);
120         section.headerView = v;
121       }
122       else
123         section.headerView = header;
124     }
125     
126     this.tableData[this.tableData.length] = section;
127   }
