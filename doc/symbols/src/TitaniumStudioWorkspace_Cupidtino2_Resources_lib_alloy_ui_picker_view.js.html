1 var used = [Ti.UI.createPicker, Ti.UI.createPickerColumn, Ti.UI.createPickerRow];
  2 
  3 Alloy.UI.PickerView = new JS.Class(Alloy.View, {
  4   initialize: function() {
  5     this.callSuper();
  6     this.window.backgroundColor = 'stripped';
  7     //this.window.modal = false;
  8     //this.window.navBarHidden = true;
  9     this.name = 'picker';
 10     var picker = Ti.UI.createPicker({top: 0, selectionIndicator: true});
 11     this.picker = picker;
 12     
 13     this.view.add(picker);
 14   },
 15   
 16   render: function(title, items /* array of arrays of strings*/, infoLineFormat, initialValues) {
 17     this.window.title = title;
 18     this.title = title;
 19     this.infoLineFormat = infoLineFormat;
 20     var _this = this;
 21     
 22     this.picker.addEventListener('change', function(e) {
 23       _this.selectedValue = e.selectedValue || _this.picker.value;
 24       _this.updateInfoLine();
 25       _this.picker.fireEvent('app:picker_view:change', {pickerView: _this, columnIndex: e.columnIndex, rowIndex: e.rowIndex, selectedValue: _this.selectedValue});
 26 
 27       if (_this.autoCommit) {
 28          _this.picker.fireEvent('app:picker_view:done', {pickerView: _this, selectedValue: _this.selectedValue});       
 29       }
 30     });
 31     
 32     var columns = [];
 33     
 34     // If items is a hash, it indicates a builtin picker type (e.g. date)
 35     if (typeof(items) == 'object' && items.type) {
 36       this.picker.type = items.type;
 37       if (items.minDate)
 38         this.picker.minDate = items.minDate;   
 39       _this.picker.value = initialValues;   
 40     }
 41     // Otherwise this is a custom picker. items is 2d array with columns and row values
 42     else {
 43       // Create all columns and rows
 44       for (var columnIndex in items) {
 45         var column = Ti.UI.createPickerColumn();
 46         
 47         for (var itemIndex in items[columnIndex]) {
 48           column.addRow(Ti.UI.createPickerRow({title:items[columnIndex][itemIndex]}));
 49         }
 50         columns.push(column);
 51       }
 52   
 53       this.picker.add(columns);
 54   
 55       // Select specified items or first items in all columns
 56       this.window.addEventListener('open', function() {
 57         if (initialValues && initialValues.length > 0) {
 58           var columnIndex = 0;
 59           for (var i in initialValues) {
 60             var initialValueIndex = items[columnIndex].indexOf(initialValues[i]);
 61             _this.picker.setSelectedRow(columnIndex, initialValueIndex, true);
 62             info("selecting column " + columnIndex + " row " + initialValueIndex);
 63             columnIndex++;
 64           }
 65         }
 66         else {
 67           for (var columnIndex in items) {
 68             _this.selectedValue.push(items[columnIndex][0]);
 69             _this.picker.setSelectedRow(columnIndex, 0, false);
 70           }
 71         }      
 72       });
 73     }
 74     
 75     _this.selectedValue = [];
 76         
 77     // Info line
 78     this.infoLine = new Label({color: '#000', textAlign: 'center', left: 8, right: 8, top: 224, height: 57, font: {fontSize: 19, fontWeight: 'bold'}});
 79     this.view.add(this.infoLine);
 80     this.updateInfoLine();
 81     
 82     // Nav bar buttons
 83     if (!_this.nav) {
 84       var cancelButton = new Button({
 85         title : _this.autoCommit ? 'Close' : 'Cancel',
 86         style: Ti.UI.iPhone.SystemButtonStyle.DONE
 87       });
 88       cancelButton.addEventListener('click', function(e) {
 89         _this.close({animated: true});
 90       })
 91       
 92       this.window.leftNavButton = cancelButton;      
 93     }
 94     
 95     // Present a done (i.e. save) button unless we want to auto commit changes
 96     if (!_this.autoCommit) {
 97       var doneButton = new Button({
 98         title : 'Done',
 99         style: Ti.UI.iPhone.SystemButtonStyle.DONE
100       });
101       
102       doneButton.addEventListener('click', function(e) {
103         _this.picker.fireEvent('app:picker_view:done', {pickerView: _this, selectedValue: _this.selectedValue});
104         if (_this.nav)
105           _this.nav.close(_this.window);
106         else
107           _this.window.close();
108       })
109       this.window.rightNavButton = doneButton;
110     }
111   },
112   
113   updateInfoLine: function() {
114     if (typeof(this.infoLineFormat) == 'string') {
115       var message = this.infoLineFormat;
116       message = message.replace(/\{title\}/g, this.window.title);
117       for (var index in this.selectedValue) {
118         message = message.replace('{' + index + '}', this.selectedValue[index]);
119       }
120       
121       this.infoLine.text = message;
122     }
123     else if  (typeof(this.infoLineFormat) == 'function') {
124       message = this.infoLineFormat(this.selectedValue);
125       this.infoLine.text = message;
126     }
127   },
128   
129   analyticsPageUrl: function() {
130     return 'picker?title=' + this.window.title;
131   }
132 });
