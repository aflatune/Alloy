1 Alloy.UI.TableViewPullToRefresh = function(tableView, backgroundColor, color, borderColor) {
  2   if (typeof(backgroundColor) == 'undefined') backgroundColor = "#e2e7ed";
  3   if (typeof(color) == 'undefined') color = "#576c89";
  4   if (typeof(borderColor) == 'undefined') borderColor = color;
  5   
  6   function formatDate(){
  7     var date = new Date();
  8     var datestr = date.getMonth()+'/'+date.getDate()+'/'+date.getFullYear();
  9     if (date.getHours()>=12)
 10     {
 11       datestr+=' '+(date.getHours()==12 ? date.getHours() : date.getHours()-12)+':'+date.getMinutes()+' PM';
 12     }
 13     else
 14     {
 15       datestr+=' '+date.getHours()+':'+date.getMinutes()+' AM';
 16     }
 17     return datestr;
 18   }
 19   
 20   var border = Ti.UI.createView({
 21     backgroundColor:borderColor,
 22     height:2,
 23     bottom:0
 24   });
 25   
 26   var tableHeader = Ti.UI.createView({
 27     backgroundColor:backgroundColor,
 28     width:320,
 29     height:60
 30   });
 31   
 32   // fake it til ya make it..  create a 2 pixel
 33   // bottom border
 34   tableHeader.add(border);
 35   
 36   var arrow = Ti.UI.createView({
 37     backgroundImage:"../images/whiteArrow.png",
 38     width:23,
 39     height:60,
 40     bottom:10,
 41     left:20
 42   });
 43   
 44   var statusLabel = Ti.UI.createLabel({
 45     text:"Pull to reload",
 46     left:55,
 47     width:200,
 48     bottom:30,
 49     height:"auto",
 50     color:color,
 51     textAlign:"center",
 52     font:{fontSize:13,fontWeight:"bold"},
 53     shadowColor:"#999",
 54     shadowOffset:{x:0,y:1}
 55   });
 56   
 57   var lastUpdatedLabel = Ti.UI.createLabel({
 58     text:"Last Updated: "+formatDate(),
 59     left:55,
 60     width:200,
 61     bottom:15,
 62     height:"auto",
 63     color:color,
 64     textAlign:"center",
 65     font:{fontSize:12},
 66     shadowColor:"#999",
 67     shadowOffset:{x:0,y:1}
 68   });
 69   
 70   var actInd = Titanium.UI.createActivityIndicator({
 71     left:20,
 72     bottom:13,
 73     width:30,
 74     height:30
 75   });
 76   
 77   tableHeader.add(arrow);
 78   tableHeader.add(statusLabel);
 79   tableHeader.add(lastUpdatedLabel);
 80   tableHeader.add(actInd);
 81   
 82   tableView.headerPullView = tableHeader;
 83   
 84   
 85   var pulling = false;
 86   var reloading = false;
 87   
 88   tableView.beginReloading = function() {
 89     info("**** Implement tableView.beginReloading and call tableView.endReloading when done reloading ****");
 90     tableView.endReloading();
 91   }
 92   
 93   tableView.endReloading = function() {
 94     // when you're done, just reset
 95     tableView.setContentInsets({top:0},{animated:true});
 96     reloading = false;
 97     lastUpdatedLabel.text = "Last Updated: "+formatDate();
 98     statusLabel.text = "Pull down to refresh...";
 99     actInd.hide();
100     arrow.show();
101   }
102   
103   tableView.addEventListener('scroll',function(e)
104   {
105     var offset = e.contentOffset.y;
106     if (offset <= -65.0 && !pulling)
107     {
108       var t = Ti.UI.create2DMatrix();
109       t = t.rotate(-180);
110       pulling = true;
111       arrow.animate({transform:t,duration:180});
112       statusLabel.text = "Release to refresh...";
113     }
114     else if (pulling && offset > -65.0 && offset < 0)
115     {
116       pulling = false;
117       var t = Ti.UI.create2DMatrix();
118       arrow.animate({transform:t,duration:180});
119       statusLabel.text = "Pull down to refresh...";
120     }
121   });
122   
123   tableView.addEventListener('scrollEnd',function(e)
124   {
125     if (pulling && !reloading && e.contentOffset.y <= -65.0)
126     {
127       reloading = true;
128       pulling = false;
129       arrow.hide();
130       actInd.show();
131       statusLabel.text = "Reloading...";
132       tableView.setContentInsets({top:60},{animated:true});
133       arrow.transform=Ti.UI.create2DMatrix();
134       tableView.beginReloading();
135     }
136   });
137 
138 }
