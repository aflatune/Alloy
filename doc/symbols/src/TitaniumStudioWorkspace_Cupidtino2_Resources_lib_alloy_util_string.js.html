1 String.prototype.toCamelCase = function() {
  2   return this.toString()
  3     .replace(/([A-Z]+)/g, function(m,l){
  4       return l.substr(0,1).toUpperCase() + l.toLowerCase().substr(1,l.length);
  5     })
  6     .replace(/[\-_\s](.)/g, function(m, l){
  7       return l.toUpperCase();
  8     });
  9 };

 11 String.prototype.toPascalCase = function() {
 12   return this.toString()
 13     .replace(/([a-zA-Z]+)/g, function(m,l){
 14       return l.substr(0,1).toUpperCase() + l.toLowerCase().substr(1,l.length);
 15     })
 16     .replace(/[\-_\s](.)/g, function(m, l){
 17       return l.toUpperCase();
 18     });
 19 };

 21 String.prototype.underscore = function() {
 22   return this.toString()
 23 						 .replace(/::/g, '/')
 24              .replace(/([A-Z]+)([A-Z][a-z])/g, '$1_$2')
 25              .replace(/([a-z\d])([A-Z])/g, '$1_$2')
 26              .replace(/-/g, '_')
 27              .toLowerCase();
 28 }

 30 String.prototype.dasherize = function() {
 31   return this.underscore().replace(/_/g, '-');
 32 }

 34 String.prototype.trim = function(chars) {
 35   return this.ltrim(this.rtrim(this.toString(), chars), chars);
 36 }
 37 
 38 String.prototype.ltrim = function(chars) {
 39   chars = chars || "\\s";
 40   return this.toString().replace(new RegExp("^[" + chars + "]+", "g"), "");
 41 }
 42  
 43 String.prototype.rtrim = function(chars) {
 44   chars = chars || "\\s";
 45   return this.toString().replace(new RegExp("[" + chars + "]+$", "g"), "");
 46 }

 48 String.prototype.isValidEmail = function() {
 49   var re = /^(([^<>()[\]\\.,;:\s@\"]+(\.[^<>()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
 50 
 51   return re.test(this.toString());
 52 }
