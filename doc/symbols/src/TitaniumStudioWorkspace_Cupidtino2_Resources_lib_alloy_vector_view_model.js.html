1 Alloy.VectorViewModel = new JS.Class(Alloy.ViewModel, {
  2   initialize: function(view) {

  5     // Vector view model stores an array of objects

  7 
  8     // [Optional] page number to load rows progressively
  9     this.rowsPerPage = 10;
 10     
 11     // Name of the unique id attribute, same as database table primary key

 13     this.idAttributeName = 'id';
 14   },

 16   fetch: function(params) {
 17     if (params.page == null)
 18       params.page = Math.ceil((this.data.length / this.rowsPerPage) + 1);
 19       
 20     params.rowsPerPage = this.rowsPerPage;
 21     
 22     if (params.type == 'reload') {
 23       params.page = 1; // Load first page again
 24       //this.oldData = this.data;
 25       //this.data = [];
 26       params.rowsPerPage = this.data.length || this.rowsPerPage;
 27       
 28       // If last time not enough items got returned, now ask for more
 29       if (params.rowsPerPage < this.rowsPerPage)
 30         params.rowsPerPage = this.rowsPerPage;
 31       
 32       //info("params.rowsPerPage = " + params.rowsPerPage);
 33     }
 34    
 35     this.callSuper(params);
 36   },

 38   dataReady: function(data, params) {
 39     
 40     // The view is interested in all data together
 41     if (this.view.dataReady) {
 42       //this.view.dataReady(data, params, this);
 43     }
 44     
 45     // Or as diff updates and append calls
 46     else {
 47       this.view.beginUpdate();
 48   
 49       // If starting from empty data, repaint the table with new data
 50       //if (this.data.length == 0) {

 54       // otherwise do diff updates

 59         // Diff apply changes
 60         this.diffUpdate(this.data, data, (params.page - 1) * params.rowsPerPage, params.type == 'reload');
 61       }

 63         // Append
 64         for (var i = 0; i < data.length; i++) {
 65           var newDataItem = data[i];
 66           var row = this.view.createRow(newDataItem, this, i);

 70       }

 72     }
 73   },
 74   
 75   
 76   lookupID: function(id, data) {
 77     for (var i = 0; i < data.length; i++) {
 78       if (data[i][this.idAttributeName] == id)
 79         return i;
 80     }
 81     
 82     return -1;
 83   },
 84   
 85   diffUpdate: function(oldData, newData, newDataStartIndex, fullReload) {
 86     //info('Starting diff update');
 87     
 88     var oldDataCount = oldData.length;
 89     var exhaustedOldData = false;
 90     
 91     //info('oldData length: ' + oldData.length);
 92     //info('newData length: ' + newData.length);
 93     
 94     for (var i = 0; i < newData.length; i++) {
 95       var newDataItem = newData[i];
 96       if (i >= oldDataCount - newDataStartIndex)
 97         exhaustedOldData = true;
 98       
 99       var oldDataIndex = i + newDataStartIndex;
100       var oldDataItem = oldData[oldDataIndex];
101       var row = this.view.createRow(newDataItem, this, i);
102       
103       // If no more items in the old data, just append items from new data
104       if (exhaustedOldData) {
105         this.view.appendRow(row, this);
106         oldData.push(newDataItem);
107       }
108       else {
109         var newID = newDataItem[this.idAttributeName];
110         var oldID = oldDataItem[this.idAttributeName];
111         // If item id is same, 
112         if (newID == oldID) {
113           // and contents have changed, update the item
114           if (this.sameObjects(newDataItem, oldDataItem)) {
115             //info('Skipping row ' + oldDataIndex);
116           }
117           else {
118             this.view.updateRow(oldDataIndex, row, this);
119             oldData[oldDataIndex] = newDataItem;
120           }  
121         }
122         else {
123           // If the item existed elsewhere in the old data, then remove it
124           var movingUp = false;
125           var oldItemPosition = this.lookupID(newID, oldData);
126           if (oldItemPosition >= 0) {
127             movingUp = true;
128             this.view.deleteRow(oldItemPosition, this, movingUp);
129             oldData.splice(oldItemPosition, 1); // remove from position i
130             
131           }
132           
133           // Insert item at new position
134           this.view.insertRow(oldDataIndex, row, this, movingUp);
135           oldData.splice(oldDataIndex, 0, newDataItem);  // insert at position i
136         }
137       }
138     }
139     
140     // Remove extra rows left over at the bottom
141     if (fullReload) {
142       if (newData.length + newDataStartIndex < oldData.length) {
143         for (var i = oldData.length - 1; i >= newData.length + newDataStartIndex; i --) {
144           this.view.deleteRow(i, this);
145           oldData.splice(i, 1);
146         }
147       }
148       
149     }
150 
151     
152   },
153

156   },
