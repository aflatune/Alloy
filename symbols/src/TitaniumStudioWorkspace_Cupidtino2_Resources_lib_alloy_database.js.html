1 Alloy.Database = new JS.Class({
  2   extend: {
  3     getInstance: function() {
  4       if (typeof(Alloy.Database.instance) == 'undefined')
  5         Alloy.Database.instance = new Alloy.Database();
  6         
  7       return Alloy.Database.instance;
  8     }
  9   },
 10   
 11   db_name: "ti_store",
 12 
 13   initialize: function() {
 14     Titanium.API.info("Initializing DB");
 15     this.db = Titanium.Database.open(this.db_name);
 16 
 17     Titanium.API.log("debug", "initialize_db");
 18     this.db.execute('CREATE TABLE IF NOT EXISTS schema_migrations (version text NOT NULL PRIMARY KEY)');
 19     this.db.execute('CREATE TABLE IF NOT EXISTS view_model_cache (url text NOT NULL PRIMARY KEY, data text)');
 20 
 21     this.seed_db();
 22   },
 23 
 24   seed_db: function() {
 25     Titanium.API.log("debug", "seed_db");
 26     // use this hook to initialize the database
 27   },
 28   
 29   loadMigrations: function()
 30   {
 31     var results = {};
 32 
 33     //Get TODOs from database
 34     var resultSet = this.db.execute("SELECT * FROM schema_migrations");
 35     while (resultSet.isValidRow()) {
 36       var row;
 37       for (var i = 0; i < resultSet.fieldCount(); i++) {
 38         row = resultSet.field(i);
 39       }
 40       results[row] = row;
 41       resultSet.next();
 42     }
 43     resultSet.close();
 44 
 45     return results;
 46   },
 47   
 48   migrate_db: function() {
 49     var migrations = this.loadMigrations();
 50 
 51     try {
 52       var migration_dir = Titanium.Filesystem.getFile(Titanium.Filesystem.resourcesDirectory, "db", "migrations");
 53   
 54       var files = migration_dir.getDirectoryListing();
 55       for (var i = 0; i < files.length; i++) {
 56         var file = files[i];
 57   			var parts = file.split(".");
 58   			if(parts[1] == 'migration') {
 59           var version = parts[0];
 60   
 61   				if (!migrations[version])
 62   				{
 63   			    Ti.API.info("Running migration... " + version);
 64   			    Ti.include("/db/migrations/"  + file);
 65   				}							
 66   			}
 67       }
 68     }
 69     catch(e) {}
 70   },
 71   
 72   migrate: function(version, sql) {
 73     this.db.execute(sql);
 74     this.db.execute("INSERT INTO schema_migrations VALUES (?)", version);
 75   }
 76 });
 77 Alloy.Database.getInstance().migrate_db();
 78 
 79 function migrate(version, sql) {
 80   Alloy.Database.getInstance().migrate(version, sql);
