1 var used = [Titanium.UI.createSwitch, Titanium.UI.createTextArea];
  2 
  3 Alloy.UI.FormBuilder = new JS.Class({
  4   initialize: function(view) {
  5     this.view = view;
  6     this.rowCount = 0;
  7 
  8     this.simpleRowConfig = {
  9       control: {
 10         left: '4%',
 11         width: '96%',
 12         className: 'formBuilderSimpleRowControl'
 13       }
 14     },
 15     
 16     this.splitRowConfig = {
 17       leftControl: {
 18         left: '4%',
 19         width: '36%',
 20         className: 'formBuilderSplitRowLeftControl'
 21       },
 22       rightControl: {
 23         left: '40%',
 24         width: '56%',
 25         className: 'formBuilderSplitRowRightControl'
 26       }
 27     };
 28         
 29     this.tableData = [];
 30   },
 31   
 32   createRow : function(height) {
 33     this.rowCount++;
 34     return Ti.UI.createTableViewRow({
 35       height : height,
 36       allowsSelection:false,
 37       separatorStyle: 0
 38     });
 39   },
 40 
 41   render: function() {
 42     this.tableView = Ti.UI.createTableView({data:this.tableData, style:Titanium.UI.iPhone.TableViewStyle.GROUPED});
 43     this.view.add(this.tableView);
 44     return this.tableView;
 45   },
 46 
 47   createButton : function(height, title, color, backgroundColor) {
 48     var row = this.createRow();
 49     if (backgroundColor)
 50       row.backgroundColor = backgroundColor;
 51     row.height = height;
 52     row.add(new Label({
 53       text: title, 
 54       width: '100%', 
 55       font: {fontSize: 14, fontWeight: 'bold'},
 56       textAlign: 'center',
 57       top: 0,
 58       color: color
 59     }));
 60     this.currentSection().add(row);
 61     
 62     return row;
 63   },
 64   
 65   createLabel: function(text) {
 66     var row = this.createRow();
 67     row.height = 'auto';
 68     
 69     row.hasChild = true;
 70     var label = new Label({
 71       text: text, 
 72       font: {fontSize: 14},
 73       textAlign: 'left',
 74       left: 8,
 75       right: 8,
 76       height: 40
 77     });
 78     row.add(label);
 79     this.currentSection().add(row);
 80     
 81     return ({row: row, label: label});
 82   },
 83   
 84   _addSplitRowLabel: function(row, labelText, height) {
 85     var label = new Label({
 86       className: this.splitRowConfig.leftControl.className,
 87       top : 0,
 88       left : this.splitRowConfig.leftControl.left,
 89       width : this.splitRowConfig.leftControl.width,
 90       height : height,
 91       text : labelText
 92     });
 93       
 94     row.add(label); 
 95     return label;
 96   },
 97 
 98   createTextField : function(height, labelText, hintText, text, keyboardType, returnKeyType, backgroundColor) {
 99     return this.createTextFieldInternal(true, height, labelText, hintText, text, keyboardType, returnKeyType, backgroundColor);
100   },
101 
102   createTextArea : function(height, labelText, hintText, text, keyboardType, returnKeyType, backgroundColor) {
103     return this.createTextFieldInternal(false, height, labelText, hintText, text, keyboardType, returnKeyType, backgroundColor);
104   },
105     
106   createTextFieldInternal : function(textFieldType, height, labelText, hintText, text, keyboardType, returnKeyType, backgroundColor) {
107     var row = this.createRow(height);
108     if (backgroundColor)
109       row.backgroundColor = backgroundColor;
110       
111     var textFieldConfig;
112 
113     if(labelText) {
114       this._addSplitRowLabel(row, labelText, height);
115       textFieldConfig = this.splitRowConfig.rightControl;
116     }
117     else { 
118       textFieldConfig = this.simpleRowConfig.control;
119     }
120 
121     var textControl = textFieldType ? TextField : TextArea;
122     
123     var textField = new textControl({
124       className: this.splitRowConfig.rightControl.className,
125       top : 4,
126       left : textFieldConfig.left,
127       width : textFieldConfig.width,
128       height : height - 8,
129       borderStyle : Titanium.UI.INPUT_BORDERSTYLE_NONE,
130       hintText : hintText,
131       clearButtonMode : Ti.UI.INPUT_BUTTONMODE_ALWAYS,
132       value : text
133     });
134 
135     if (keyboardType)
136       textField.keyboardType = keyboardType;
137       
138     if (returnKeyType)
139       textField.returnKeyType = returnKeyType;
140 
141     row.add(textField);
142 
143     this.currentSection().add(row);
144 
145     return textField;
146   },
147 
148   createSwitch: function(height, label, value) {
149     var row = this.createRow(height);
150     var leftControl = this._addSplitRowLabel(row, label, height);
151     row.add(leftControl);
152     
153     var rightControl = new Switch({
154       className: this.splitRowConfig.rightControl.className,
155       right: this.splitRowConfig.leftControl.left,
156       value: value      
157     });
158     row.add(rightControl);
159     this.currentSection().add(row);
160     
161     return ({row:row, leftControl:leftControl, rightControl:rightControl});
162   },
163   
164   createSplitRowLabels : function(height, label1Text, label2Text) {
165     var row = this.createRow(height);
166     var label1 = this._addSplitRowLabel(row, label1Text, height);
167     row.add(label1);
168     
169     var label2 = new Label({
170       className: this.splitRowConfig.rightControl.className,
171       left : this.splitRowConfig.rightControl.left,
172       width : this.splitRowConfig.rightControl.width,
173       text : label2Text
174     });
175     row.add(label2);
176 
177     this.currentSection().add(row);
178 
179     return ({row:row, leftControl:label1, rightControl:label2});
180   },
181   
182   currentSection: function() {
183     if (this.tableData.length == 0)
184       this.startNewSection();
185       
186     return this.tableData[this.tableData.length-1];
187   },
188   
189   startNewSection: function(title) {
190     var section = Ti.UI.createTableViewSection();
191     section.headerTitle = title;
192     this.tableData[this.tableData.length] = section;
193     return section;
194   }
195 })
