 29 /**
 30  * @namespace Fallback for CSS advanced media query
 31  * @class
 32  * @since 1.0 - 2010-09-10
 33  * @version 1.0 - 2010-09-10
 34  */
 35 wbos.CssTools.MediaQueryFallBack = ( function() {
 36 	var config = {
 37 		cssScreen: "/css/screen.css",
 38 		cssHandheld: "/css/handheld.css",
 39 		mobileMaxWidth: 660,
 40 		testDivClass: "cssLoadCheck",
 41 		dynamicCssLinkId: "DynCssLink",
 42 		resizeDelay: 30
 43 	}
 44 	var noMediaQuery = false;
 45 	var delay;
 46 	var currentCssMediaType;
 47 
 48 	// Adding events to elements in the DOM without overwriting it
 49 	function addEvent(element, newFunction, eventType) {
 50 		var oldEvent = eval("element." + eventType);
 51 		var eventContentType = eval("typeof element." + eventType)
 52 		
 53 		if ( eventContentType != 'function' ) {
 54 			eval("element." + eventType + " = newFunction")
 55 		} else {
 56 			eval("element." + eventType + " = function(e) { oldEvent(e); newFunction(e); }")
 57 		}
 58 	}
 59 	
 60 	// Get the the inner width of the browser window
 61 	function getWindowWidth() {
 62 		if (window.innerWidth) {
 63 			return window.innerWidth;
 64 		} else if (document.documentElement.clientWidth) {
 65 			return document.documentElement.clientWidth;
 66 		} else if (document.body.clientWidth) {
 67 			return document.body.clientWidth;
 68 		} else{
 69 			return 0;	
 70 		}
 71 	}
 72 	
 73 	function addCssLink(cssHref) {
 74 		var cssNode = document.createElement('link');
 75 		var windowWidth;
 76 		cssNode.type = 'text/css';
 77 		cssNode.rel = 'stylesheet';
 78 		cssNode.media = 'screen, handheld, fallback';
 79 		cssNode.href = cssHref;
 80 		document.getElementsByTagName("head")[0].appendChild(cssNode);
 81 	}
 82 	
 83 
 84 
 85 	/* Start public */
 86 	return {
 87   		/**
 88 		 * Adds link to CSS in the head if no CSS is loaded
 89 		 *
 90 		 * @since 1.0 - 2010-08-21
 91 		 * @version 1.0 - 2010-08-21
 92 		 * @param {String|Object} cssScreen URL to CSS file for larger screens
 93 		 * @param {String|Object} cssHandheld URL to CSS file for smaller screens
 94 		 * @param {Number} mobileMaxWidth Maximum  width for handheld devices
 95 		 * @example
 96 		 * wbos.CssTools.MediaQueryFallBack.LoadCss(['screen.css', 'screen2.css'], 'mobile.css', 480)
 97 		 */
 98 		LoadCss: function(cssScreen, cssHandheld, mobileMaxWidth) {
 99 			// Set config values
100 			if (typeof(cssScreen) != "undefined") {
101 				config.cssScreen = cssScreen;	
102 			}
103 			if (typeof(cssHandheld) != "undefined") {
104 				config.cssHandheld = cssHandheld;	
105 			}
106 			if (typeof(mobileMaxWidth) != "undefined") {
107 				config.mobileMaxWidth = mobileMaxWidth;	
108 			}
109 			
110 			// Check if CSS is loaded
111 			var cssloadCheckNode = document.createElement('div');
112 			cssloadCheckNode.className = config.testDivClass;
113 			document.getElementsByTagName("body")[0].appendChild(cssloadCheckNode);
114 			if (cssloadCheckNode.offsetWidth != 100 && noMediaQuery == false) {
115 				noMediaQuery = true;
116 			}
117 			cssloadCheckNode.parentNode.removeChild(cssloadCheckNode)
118 			
119 			if (noMediaQuery == true) {
120 				// Browser does not support Media Queries, so JavaScript will supply a fallback 
121 				var cssHref = "";
122 				
123 				// Determines what CSS file to load
124 				if (getWindowWidth() <= config.mobileMaxWidth) {
125 					cssHref = config.cssHandheld;
126 					newCssMediaType = "handheld";
127 				} else {
128 					cssHref = config.cssScreen;
129 					newCssMediaType = "screen";
130 				}
131 				
132 				// Add CSS link to <head> of page
133 				if (cssHref != "" && currentCssMediaType != newCssMediaType) {
134 					var currentCssLinks = document.styleSheets
135 					for (var i = 0; i < currentCssLinks.length; i++) {
136 						for (var ii = 0; ii < currentCssLinks[i].media.length; ii++) {
137 							if (typeof(currentCssLinks[i].media) == "object") {
138 								if (currentCssLinks[i].media.item(ii) == "fallback") {
139 									currentCssLinks[i].ownerNode.parentNode.removeChild(currentCssLinks[i].ownerNode)
140 									i--
141 									break;
142 								}
143 							} else {
144 								if (currentCssLinks[i].media.indexOf("fallback") >= 0) {
145 									currentCssLinks[i].owningElement.parentNode.removeChild(currentCssLinks[i].owningElement)
146 									i--
147 									break;
148 								}
149 							}
150 						}
151 					}
152 					if (typeof(cssHref) == "object") {
153 						for (var i = 0; i < cssHref.length; i++) {
154 							addCssLink(cssHref[i])
155 						}
156 					} else {
157 						addCssLink(cssHref)
158 					}
159 										
160 					currentCssMediaType = newCssMediaType;
161 				}
162 
163 				
164 				// Check screen size again if user resizes window 
165 				addEvent(window, wbos.CssTools.MediaQueryFallBack.LoadCssDelayed, 'onresize')
166 			}
167 		},
168 		
169   		/**
170 		 * Runs LoadCSS after a short delay
171 		 *
172 		 * @since 1.0 - 2010-08-21
173 		 * @version 1.0 - 2010-08-21
174 		 * @example
175 		 * wbos.CssTools.MediaQueryFallBack.LoadCssDelayed()
176 		 */
177 		LoadCssDelayed: function() {
178 			clearTimeout(delay);
179 			delay = setTimeout( "wbos.CssTools.MediaQueryFallBack.LoadCss()", config.resizeDelay)
180 		}
181 		
182 	}
183 	/* End public */
184 })();

202 		 * Adds a function to an event of a single element
203 		 *
204 		 * @since 1.0 - 2010-02-23
205 		 * @version 1.0 - 2010-02-23
206 		 * @param {Object} element The element on which the event is placed
207 		 * @param {Function} newFunction The function that has to be linked to the event
208 		 * @param {String} eventType Name of the event
209 		 * @example
210 		 * wbos.Events.AddEvent( document.getElementById('elementId'), functionName, "onclick" )
211 		 */
212 		AddEvent: function( element, newFunction, eventType ) {
213 			var oldEvent = eval("element." + eventType);
214 			var eventContentType = eval("typeof element." + eventType)
215 			
216 			if ( eventContentType != 'function' ) {
217 				eval("element." + eventType + " = newFunction")
218 			} else {
219 				eval("element." + eventType + " = function(e) { oldEvent(e); newFunction(e); }")
220 			}
221 		}
222 	}

244 /**
245  * @namespace Enables filtering in class lists
246  * @class
247  * @since 1.0 - 2010-11-08
248  * @version 1.0 - 2010-11-08
249  */
250 codeview.classFilter = ( function() {
251 	function onkeyup_ClassFilter() {
252 		var listItems
253 		var search = document.getElementById('ClassFilter').value
254 		search = search.toLowerCase()
255 		if (document.getElementById('ClassList')) {
256 			listItems = document.getElementById('ClassList').getElementsByTagName('li')
257 			filterList(listItems, search)
258 		}
259 		if (document.getElementById('ClassList2')) {
260 			listItems = document.getElementById('ClassList2').getElementsByTagName('li')
261 			filterList(listItems, search)
262 		}
263 		if (document.getElementById('FileList')) {
264 			listItems = document.getElementById('FileList').getElementsByTagName('li')
265 			filterList(listItems, search)
266 		}
267 		if (document.getElementById('MethodsListInherited')) {
268 			var links = document.getElementById('MethodsListInherited').getElementsByTagName('a')
269 			var linksSelected = new Array()
270 			for (var i=0; i < links.length; i++) {
271 				if (links[i].parentNode.parentNode.tagName == "DD") {
272 					linksSelected.push(links[i])
273 				}
274 			}
275 			filterList(linksSelected, search)
276 		}
277 		if (document.getElementById('MethodsList')) {
278 			listItems = document.getElementById('MethodsList').getElementsByTagName('tbody')[0].getElementsByTagName('tr')
279 			filterList(listItems, search, document.getElementById('MethodDetail').getElementsByTagName('li'))
280 		}
281 	}
282 	
283 	function filterList(listItems, search, relatedElements) {
284 		var itemContent = ""
285 		for (var i=0; i < listItems.length; i++) {
286 			itemContent = listItems[i].textContent||listItems[i].innerText
287 			if (itemContent != undefined) {
288 				itemContent = itemContent.toLowerCase()
289 				itemContent = itemContent.replace(/\s/g, "")
290 				if (itemContent.indexOf(search) >= 0 || itemContent == "") {
291 					listItems[i].style.display = ""
292 				} else {
293 					listItems[i].style.display = "none"
294 				}
295 				if (relatedElements != null) {
296 					filterRelatedList(listItems[i], search, relatedElements)
297 				}
298 			}
299 		}
300 	}
301 	
302 	function filterRelatedList(listItem, search, relatedElements) {
303 		var itemIndex = parseInt(listItem.className.replace('item', ''))
304 		if (itemIndex <= relatedElements.length) {
305 			if (relatedElements[itemIndex].className == "item"+ itemIndex) {
306 				relatedElements[itemIndex].style.display = listItem.style.display
307 			}
308 		}
309 	}
310

315 	/* Start public */
316 	return {
317 		Init: function() {
318 			wbos.Events.AddEvent(
319 				document.getElementById('ClassFilter'),
320 				onkeyup_ClassFilter,
321 				"onkeyup"
322 			)					
323 		}
324 	}
325 	/* End public */
326 })();
