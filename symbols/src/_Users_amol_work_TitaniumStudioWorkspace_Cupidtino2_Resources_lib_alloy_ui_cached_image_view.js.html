1 Alloy.UI.CachedImageView = new JS.Class({
  2   extend: {
  3     CachePolicy: {
  4       Disabled: 1,        // No caching, i.e. fetch directly from url, no offline support -- same as normal ImageView
  5       OfflineOnly: 2,     // Use cached image only when offline. When online, fetch from url and store in cache.
  6       CacheAndServe: 3},  // If cache hit, do not fetch again from url. Useful when images are immutable on server.
  7   },
  8 
  9   initialize: function(params) {
 10     this.imageView = new ImageView(params);
 11     this.cachePolicy = Alloy.UI.CachedImageView.CachePolicy.CacheAndServe;
 12     
 13     // Cache write
 14     var _this = this;
 15     this.imageView.addEventListener("load", function(e) {
 16       if (_this.cachePolicy == Alloy.UI.CachedImageView.CachePolicy.Disabled)
 17         return;
 18         
 19       var filename = _this.urlToFilename(e.source.image);
 20       if (filename) {
 21          
 22         var f = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory, "cache/" + filename /*e.source.image*/);
 23         if (!f.exists() || 
 24            (_this.cachePolicy == Alloy.UI.CachedImageView.CachePolicy.OfflineOnly)) {
 25           // Make sure the directory exists
 26           var dir = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory, "cache");
 27           dir.createDirectory();
 28           
 29           // Write out the file
 30           var imageData = e.source.toImage();
 31           f.write(imageData);
 32         }
 33       }
 34     });
 35   },
 36   
 37   // Cache read
 38   setImage: function(url) {
 39     if (this.cachePolicy == Alloy.UI.CachedImageView.CachePolicy.Disabled ||
 40        (this.cachePolicy == Alloy.UI.CachedImageView.CachePolicy.OfflineOnly && Ti.Network.online)) {
 41       this.imageView.image = url;
 42       return;
 43     }
 44     
 45     var filename = this.urlToFilename(url);
 46     if (filename) {
 47       var f = Titanium.Filesystem.getFile(Titanium.Filesystem.applicationDataDirectory, "cache/" + filename);
 48       if (f.exists()) {
 49         this.imageView.image = f.read();
 50       }
 51       else {
 52         this.imageView.image = url;
 53       }
 54     }    
 55   },
 56   
 57   urlToFilename: function(url) {
 58     // When url is NSUrl
 59     if(typeof(url) == 'object') url = url.absoluteString;
 60     
 61     // When url is TiBlob
 62     if (url == null) return null;
 63     
 64     return url.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/\-([a-z]+)$/, '.$1');
 65   }
 66 });
