 26 Alloy.UI.ImageButtonBar = new JS.Class({
 27   extend: {
 28     createImageButtonBar: function(params) {
 29       return new Alloy.UI.ImageButtonBar(params);
 30     }
 31   },
 32   
 33   initialize: function(params) {
 34     this.toolbarView = new ScrollView(params);
 35     this.buttons = [];
 36     this.currentButton = null;
 37     this.parentView = params.parentView;
 38   },
 39   
 40   createView: function(id) {
 41     error('Must implement ImageButtonBar::createView, called when switching to a tab for the first time');
 42   },
 43   
 44   add: function(button) {
 45     button.button.left = (button.button.width + 0) * this.buttons.length;
 46     button.backgroundView.left = button.button.left;
 47     button.button.buttonIndex = this.buttons.length;
 48     
 49     var _this = this;
 50     button.button.addEventListener('click', function(e) {
 51       var switchingView = !_this.currentButton || _this.currentButton.button.buttonIndex != e.source.buttonIndex;
 52 
 53       if (switchingView) {
 54         if (_this.currentButton) {
 55           _this.currentButton.backgroundView.backgroundColor = 'transparent';
 56           _this.currentButton.tabView.view.opacity = 0;
 57           _this.currentButton = null;
 58         }
 59   
 60         _this.currentButton = _this.buttons[e.source.buttonIndex];      
 61         _this.currentButton.backgroundView.backgroundColor = '#F9D439';
 62         if (_this.currentButton.tabView) {
 63           _this.currentButton.tabView.view.opacity = 1;
 64         }
 65       }  
 66       
 67       if (_this.currentButton.tabView) {
 68         Ti.App.fireEvent('imageButtonBar:switchToRenderedTab');
 69       }
 70       else {
 71         var tabView = _this.renderTab(_this.currentButton.id);
 72         _this.parentView.view.add(tabView.view);
 73         tabView.render();
 74 
 75         _this.currentButton.tabView = tabView;
 76       }
 77       
 78       Ti.App.fireEvent('imageButtonBar:switchToTab');
 79     });
 80     
 81     this.buttons.push(button);
 82     this.toolbarView.add(button.backgroundView);
 83     this.toolbarView.add(button.button);
 84   }
 85 });
