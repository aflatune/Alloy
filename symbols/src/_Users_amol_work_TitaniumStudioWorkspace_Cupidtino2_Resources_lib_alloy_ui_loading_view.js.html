1 // Tell the compiler which modules we are going to use; note there are no () on these!
  2 var used = [Ti.UI.createActivityIndicator];
  3 
  4 Alloy.UI.LoadingView = new JS.Class(Alloy.View, {
  5   initialize: function() {
  6     this.callSuper(true);   // Partial view
  7     this.view.zIndex = 1000;
  8     this.view.visible = false;
  9   },
 10   
 11   render: function(params) {
 12     params = params || {};
 13     
 14     var backdropHeight = params.height || 100;
 15     var backdropWidth = params.width || 200;
 16     
 17     var backdrop = new View('loadingViewBackground');
 18     backdrop.left = Ti.Platform.displayCaps.platformWidth / 2 - backdropWidth / 2;
 19     backdrop.width = backdropWidth;;
 20     backdrop.top = Ti.Platform.displayCaps.platformHeight / 2 - backdropHeight / 2;
 21     backdrop.height = backdropHeight;
 22     
 23     var loader = new ActivityIndicator('loadingViewActivityIndicator');
 24     
 25     backdrop.add(loader);
 26     
 27     this.view.add(backdrop);
 28     //this.view.add(loader);
 29     
 30     var _this = this;
 31     
 32     this.showCount = 0;
 33     Ti.App.addEventListener('app:show.loader', function() {
 34       _this.showCount++;
 35       
 36       if (_this.showCount == 1 && !_this.view.visible) {
 37         _this.view.visible = true;
 38         loader.show();
 39         
 40         var t1 = Ti.UI.create2DMatrix().scale(0.1);
 41         backdrop.transform = t1;
 42         backdrop.opacity = 0;
 43         
 44         var t2 = Ti.UI.create2DMatrix().scale(1);
 45         var a = Ti.UI.createAnimation();
 46 
 47         a.transform = t2;
 48         a.opacity = 0.9;
 49         a.duration = 200;
 50 
 51         backdrop.animate(a);
 52       }
 53     });
 54 
 55     var t3 = Ti.UI.create2DMatrix().scale(1.3);
 56     var a3 = Ti.UI.createAnimation();
 57 
 58     a3.transform = t3;
 59     a3.opacity = 0;
 60     a3.duration = 300;
 61 
 62     a3.addEventListener('complete',function() {
 63       loader.hide();
 64       _this.view.visible = false;
 65     });
 66     
 67     Ti.App.addEventListener('app:hide.loader', function() {
 68       _this.showCount--;
 69       if (_this.showCount < 0) _this.showCount = 0;
 70       
 71       if (_this.showCount == 0 && _this.view.visible) {
 72         backdrop.animate(a3);          
 73       }
 74     });
 75   }
 76 });
