  3 var used = [Ti.UI.createButton, Ti.UI.createView, Ti.UI.createLabel, Ti.UI.createTextField, Ti.UI.createTableView, Ti.UI.createTableViewRow];

  5 Alloy.UI.LoginView = new JS.Singleton(Alloy.View, {
  6   config: {
  7     color: '#fff',
  8     signupText: 'Not a member? Join for free.',
  9     barColor: '#000',
 10     emailTextField: {
 11       hint: 'Email address',
 12       validate: function(email) {
 13         if (!email.isValidEmail()) {
 14           Alloy.UI.alert("Please enter a valid email address.", "Wait!");
 15           return false;
 16         }
 17         return true;
 18       }
 19     }
 20   },
 21   
 22   initialize : function() {
 23     this.callSuper();
 24     this.window.translucent = true;
 25     this.window.barColor = this.config.barColor;
 26     this.formBuilder = new Alloy.UI.FormBuilder(this.window);
 27     this.name = 'login';
 28   },
 29   
 30   render : function() {
 31     var _this = this;
 32     this.callSuper();
 33     
 34     // Top section
 35     this.formBuilder.startNewSection();
 36 
 37     var emailTextField = this.formBuilder.createTextField(40, null, this.config.emailTextField.hint, Ti.App.Properties.getString('email'), Titanium.UI.KEYBOARD_EMAIL, Titanium.UI.RETURNKEY_NEXT, this.config.backgroundColor);
 38     emailTextField.suppressReturn = true;
 39     emailTextField.color = this.config.color;
 40     this.emailTextField = emailTextField;
 41     
 42     var passwordTextField = this.formBuilder.createTextField(40, null, "Password", Ti.App.Properties.getString('password'), Titanium.UI.KEYBOARD_ASCII, Titanium.UI.RETURNKEY_GO, this.config.backgroundColor);
 43     passwordTextField.passwordMask = true;
 44     passwordTextField.suppressReturn = true;
 45     passwordTextField.color = this.config.color;
 46     this.passwordTextField = passwordTextField;
 47 
 48     // Bottom section
 49     if (this.config.signupText) {
 50       this.formBuilder.startNewSection();
 51       var joinButton = this.formBuilder.createButton(60, this.config.signupText, this.config.color, this.config.backgroundColor);
 52       joinButton.color = this.config.color;
 53       if (_this.config.signupUrl) {
 54         joinButton.addEventListener('click', function() {
 55           Alloy.UI.WebView.show(_this.config.signupUrl);
 56         })
 57       }
 58       
 59       this.signupButton = joinButton;
 60     }
 61     
 62     // Render the form
 63     var formTable = this.formBuilder.render();
 64     formTable.top = 40;
 65 
 66     // Nav bar buttons
 67     var cancelButton = new Button('toolbarButton');
 68     cancelButton.title = 'Cancel';
 69     
 70     this.window.leftNavButton = cancelButton;
 71 
 72     var loginButton = new Button({
 73       title : 'Login',
 74       backgroundColor:'blue',
 75       className: 'toolbarButton'
 76     });
 77     this.rightNavButton = loginButton;
 78     
 79     this.window.rightNavButton = loginButton;
 80     
 81     // Toolbar
 82     /*var flexSpace = Titanium.UI.createButton({
 83       systemButton:Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
 84     });
 85     var toolbar = new Toolbar({
 86       items:[cancelButton, flexSpace, loginButton],
 87       top:0,
 88       translucent:true,
 89       barColor: '#fff',
 90       backgroundImage: '/public/images/diagonal-noise.png'
 91     });
 92 
 93     this.window.add(toolbar);
 94     */
 95    
 96     // Attach toolbar button events after window loads
 97     // http://developer.appcelerator.com/question/125494/inexplicable-bug-toolbar-button-only-listens-4-clicks
 98     //this.window.addEventListener("open", function(e) {
 99       // Prevent multiple invocation when the same window is closed and opend multiple times
100       //if (_this.toolbarButtonEventsAttached)
101       //  return;
102       
103       //this.toolbarButtonEventsAttached = true;  
104       loginButton.addEventListener('click', function(e) {
105         // Login button
106         var email = emailTextField.value.trim();
107         var password = passwordTextField.value.trim();
108         
109         if (!_this.config.emailTextField.validate(email)) {
110           emailTextField.focus();
111           return;
112         }
113   
114         if (password.length < 5) {
115           Alloy.UI.alert("Your password should be a bit longer.", "Wait!");
116           passwordTextField.focus();
117           return;
118         }
119   
120         Ti.App.fireEvent('app:login', {email: email, password: password});
121       })
122 
123       cancelButton.addEventListener('click', function(e) {
124         Ti.App.fireEvent('app:login:dismiss');
125       })
126 
127     //});
128 
129     emailTextField.addEventListener('return', function() {
130       passwordTextField.focus();
131     });
132 
133     passwordTextField.addEventListener('return', function() {
134       loginButton.fireEvent('click');
135     });
136     
137     Ti.App.fireEvent('app:login:view_rendered');
138 
139     // Show error on login failure    
140     Ti.App.addEventListener('app:login:failed', function(e) {
141       alert("Sorry, try again");
142     });
143   },
144 
145   reset: function() {
146     this.emailTextField.value = '';
147     this.passwordTextField.value = '';
148   }
149 });
150 
151 Ti.App.addEventListener('app:login:show', function(e) {
152   var view = Alloy.UI.LoginView;
153   if (!view.rendered)
154     view.render();
155   
156   view.open({modal: false, nav: 'global'});
157   //view.window.open({modal:false, animated:true, title: "Cupidtino", navBarHidden: true, transition:Ti.UI.iPhone.AnimationStyle.FLIP_FROM_RIGHT});
158 });
159 
160 Ti.App.addEventListener('app:login:dismiss', function(e) {
161   Alloy.UI.LoginView.window.close({animated:true});
162 });
163 
164 Ti.App.addEventListener('app:login:succeeded', function(e) {
165     Alloy.UI.LoginView.window.close({animated:true});
166 });
