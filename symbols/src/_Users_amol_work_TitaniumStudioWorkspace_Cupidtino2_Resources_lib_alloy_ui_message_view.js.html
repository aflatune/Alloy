  3 Alloy.UI.MessageView = new JS.Class(Alloy.View, {
  4   extend: {
  5     Align: {
  6       Left: 'Left',
  7       Right: 'Right'
  8     },
  9     
 10     Color: {
 11       White: 'White',
 12       Blue: 'Blue'
 13     }
 14   },
 15   
 16   initialize: function() {
 17     this.callSuper();
 18     this.name = "message";
 19   },
 20   
 21   render: function() {
 22     this.rowCount = 0;
 23     var layoutTable = new TableView('layoutTable');
 24     layoutTable.bottom = 80;
 25      
 26     this.messagesTable = layoutTable;
 27     this.view.add(layoutTable);   
 28   
 29     // Controls toolbar
 30     var controls = new View('messageViewControls');
 31     var controlsBG = new Label('messageViewControlsBG');
 32     controls.add(controlsBG);
 33     controls.bottom = 0;
 34     
 35     var _this = this;
 36     // Textbox
 37     var textboxContainer = new Label('messageViewTextboxContainer');
 38     var textbox = new TextArea('messageViewTextbox');
 39     this.textbox = textbox;
 40     textbox.addEventListener('focus', function(e) {
 41       var a = new Animation();
 42       a.bottom = 166;
 43       a.duration = 400;
 44       _this.view.animate(a);
 45       _this.scrollToIndex(_this.rowCount - 1);
 46     });
 47     
 48     textbox.addEventListener('blur', function(e) {
 49       var a = new Animation();
 50       a.bottom = 0;
 51       a.duration = 200;
 52       _this.view.animate(a);
 53     });
 54     controls.add(textboxContainer);
 55     controls.add(textbox);
 56     
 57     // Button
 58     var sendButton = new Button('messageViewSendButton');
 59     this.sendButton = sendButton;
 60     controls.add(sendButton);
 61     
 62     this.view.add(controls);
 63     
 64     // Dismiss keyboard when clicked elsewhere
 65     layoutTable.addEventListener('click', function() {
 66       textbox.blur();
 67     })
 68   },
 69   
 70   reset: function() {
 71     this.messagesTable.setData([]);
 72     this.rowCount = 0;
 73   },
 74   
 75   addMessage: function(text, align, color, bold) {
 76     var row = new TableViewRow('layoutTableRow');
 77     
 78     var container = new View({left: 4, right: 4, top: 0, bottom: 8, height: 'auto'});
 79     row.add(container);
 80     
 81     if (!align) align = Alloy.UI.MessageView.Align.Left;
 82     if (!color) color = Alloy.UI.MessageView.Color.White;
 83     
 84     if (align == Alloy.UI.MessageView.Align.Left)
 85       container.right = 50;
 86     else
 87       container.left = 50;
 88       
 89     var balloon = new Label('balloon' + align);
 90     balloon.backgroundImage = '/lib/alloy/ui/images/' + color + 'Balloon' + align + '.png';
 91     balloon.text = text;
 92     if (bold)
 93       balloon.font = {fontWeight: 'bold'};
 94       
 95     container.add(balloon);
 96     this.messagesTable.appendRow(row);
 97     
 98     return this.rowCount++;
 99   },
100   
101   addCustomRow: function(row) {
102     this.messagesTable.appendRow(row);
103     this.rowCount++;
104   },
105   
106   scrollToIndex: function(index) {
107     if (index == -1) index = this.rowCount - 1;
108     
109     this.messagesTable.scrollToIndex(index);
110   }
111 })
