1 Alloy.UI.TableViewDynamicUpdate = function(tableView, color, maxRows) {
  2   if (typeof(maxRows) == 'undefined') maxRows = 100;
  3   
  4 	var updating = false;
  5 	var loadingText = "Loading...";
  6 	var loadingRow = Ti.UI.createTableViewRow({title:loadingText, color: color});
  7 
  8 	tableView.endUpdate = function()
  9 	{
 10 		updating = false;
 11     var deletedRows = Alloy.UI.deleteRow(tableView, function(row) {
 12       return (row.title == loadingText);
 13     });
 14     
 15 		// just scroll down a bit to the new rows to bring them into view
 16 		//if (deletedRows.length > 0)
 17   	//	tableView.scrollToIndex(deletedRows[0],{animated:true,position:Ti.UI.iPhone.TableViewScrollPosition.BOTTOM});
 18 	}
 19 
 20 	var lastDistance = 0; // calculate location to determine direction
 21 
 22 	tableView.addEventListener('scroll',function(e)
 23 	{
 24 	  if (tableView.data.length == 0)
 25 	     return;
 26 	   
 27 	  var rows = tableView.data[0].rowCount;
 28 	  if (tableView.data[0].rowCount >= maxRows)
 29 	   return;
 30 	   
 31 		var offset = e.contentOffset.y;
 32 		var height = e.size.height;
 33 		var total = offset + height;
 34 		var theEnd = e.contentSize.height;
 35 		var distance = theEnd - total;
 36 
 37 		// going down is the only time we dynamically load,
 38 		// going up we can safely ignore -- note here that
 39 		// the values will be negative so we do the opposite
 40 		if (distance < lastDistance)
 41 		{
 42 			// adjust the % of rows scrolled before we decide to start fetching
 43 			var nearEnd = theEnd - 100; //theEnd * .75;
 44 
 45 			if (!updating && (total >= nearEnd))
 46 			{
 47 				updating = true;
 48 				tableView.appendRow(loadingRow);
 49 				
 50 				tableView.beginUpdate();
 51 			}
 52 		}
 53 		lastDistance = distance;
 54 	});
 55 }
