 51     @class Alloy View class
 52 
 53     @borrows Util.EventListener
 54     @author Amol Kelkar
 55     @since Alloy 1.0
 56     @see Alloy
 57     
 58     @constructs
 59     @param {boolean} [partial=false] Whether to instantiate the View as a partial.
 60       When true, View is created only with a view, without a window. You then add the view in another view or window.
 61   */ 
 62   initialize: function(partial) {
 63     this.view = new View();
 64     
 65     if (!partial) {
 66       this.window = Ti.UI.createWindow();
 67       $(this.window).applyStyle('Window', { className: 'viewWindow' });
 68       var _this = this;
 69       this.addEventListener(this.window, 'focus', function() {
 70         _this.trackViewShowEvent();
 71       });
 72       this.addEventListener(this.window, 'close', function() {
 73         _this.removeEventListeners();
 74       });
 75       
 76       this.window.add(this.view);
 77     }
 78   },
 79 
 80   analyticsPageUrl: function() {
 81     return (this.name || '(anonymous)');
 82   },
 83   
 84   trackViewShowEvent: function() {
 85     Alloy.analytics.trackPageview('/' + this.analyticsPageUrl());
 86   },
 87   
 88   render: function() {
 89     this.rendered = true;
 90   },
 91   
 92   open: function(params) {
 93     params = params || {};
 94     var _this = this;
 95     if (params.modal == 'simulated' && params.animated) {
 96       setTimeout(function() {
 97         _this.openWithSimulatedAnimation(params);
 98       }, 10);      
 99     }
100     else {
101       if (params.nav) {
102         if (params.nav == 'global'){
103           if (App.tabGroup && App.tabGroup.activeTab) {
104             params.nav = App.tabGroup.activeTab;
105           }
106           else
107             params.nav = true;
108         }
109 
110         // nav = true means create a new nav group
111         if (typeof(params.nav) == "boolean") {
112           var contentWindow = new Window();
113           contentWindow.navBarHidden = true;
114           var nav = Ti.UI.iPhone.createNavigationGroup({
115             window:this.window
116           });
117           this.nav = nav;
118           this.navWrapper = contentWindow;
119           
120           contentWindow.add(nav);
121           contentWindow.open({modal:true});
122         }
123         else {
124           this.nav = params.nav;
125           this.nav.open(this.window, params);
126         }
127       }
128       else {
129         this.window.open(params);
130       }
131     }
132   },
133   
134   openWithSimulatedAnimation: function(params) {
135     // Toolbar
136     var flexSpace = Titanium.UI.createButton({
137       systemButton:Titanium.UI.iPhone.SystemButton.FLEXIBLE_SPACE
138     });
139     
140     var items = [];
141     if (this.window.leftNavButton) {
142       this.window.leftNavButton.style = Ti.UI.iPhone.SystemButtonStyle.DONE;
143       items.push(this.window.leftNavButton);
144     }
145 
146     items.push(flexSpace);
147     
148     if (this.window.title) {
149       var title = new Label('windowTitle');
150       title.text = this.window.title;
151       items.push(title);
152     }
153 
154     items.push(flexSpace);
155     
156     if (this.window.rightNavButton) {
157       this.window.rightNavButton.style = Ti.UI.iPhone.SystemButtonStyle.DONE;
158       items.push(this.window.rightNavButton);
159     }
160     
161     var toolbar = new Toolbar({
162       items: items,
163       top:0,
164       translucent:true,
165       barColor: '#000'
166     });
167 
168     this.window.add(toolbar);
169     this.view.top = 40;
170     
171     this.simulatedModal = true;
172     this.window.top = Titanium.Platform.displayCaps.platformHeight - 40; //this.window.height;
173     
174     var a = new Animation();
175     params.modal = false;
176     this.window.open(params);
177     a.top = 0;
178     a.duration = 500;
179     this.window.animate(a);
180   },
181   
182   close: function(params) {
183     params = params || {};
184     var _this = this;
185     
186     var w = _this.navWrapper || _this.window;
187     
188     if (this.simulatedModal && params.animated) {
189       var a = new Animation();
190       a.top = this.window.height;
191       a.duration = 500;
192       a.addEventListener('complete', function() {
193         w.close({animated: false});
194       })
195       w.animate(a);
196     }
197     else {
198       w.close(params);
199     }
200   }
