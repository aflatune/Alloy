1 Alloy.ViewModel = new JS.Class(/** @lends Alloy.ViewModel */{
  2   extend: {
  3     CachePolicy: {
  4       Disabled: 1,        // No caching, i.e. fetch directly from web service, no offline support
  5       OfflineOnly: 2,     // Use cached data only when offline. When online, fetch from web service and store in cache.
  6       CacheAndFetch: 3,   // First use cached data, then fetch from web service and use that data. 2 calls to dataReady() if cache hit.
  7       CacheAndServe: 4},  // If cache hit, do not fetch again from web service. Useful for tracking unique app installs, etc. 
  8   },

 73     @class Alloy ViewModel class
 74 
 75     @author Amol Kelkar
 76     @since Alloy 1.0
 77     @see Alloy, Alloy.View
 78     
 79     @constructs
 80     @param {Alloy.View} [view=null] View associated with this View Model.
 81       When view is specified, view.dataReady(...) or view.onError(...) method is invoked
 82       when data is received or an error occurs, respectively. Here's the typical usage -
 83 
 84   */
 85   initialize: function(view) {
 86     this.view = view;
 87     this.data = null;
 88     this.currentRequest = null;
 89     this.cachePolicy = Alloy.ViewModel.CachePolicy.OfflineOnly;
 90     this.httpVerb = 'GET';
 91     this.showLoginViewOn403 = false;
 92   },
 93   
 94   block: function() {
 95     this.blocked = true;
 96   },
 97   
 98   unblock: function() {
 99     this.blocked = false;
100   },
101   
102   fetch: function(params) {
103     if (this.blocked)
104       return;
105     
106     var url = this.url(params);
107     //info(url);
108     
109     if (typeof(params) == 'undefined') params = {};
110     Alloy.analytics.trackEvent('alloy:view_model:' + this.name, 'fetch', url);
111 
112     // Skip reading from cache when --
113     if (
114         // getting more rows, OR
115         //params.type == 'more' ||
116         // cache policy is to never cache, OR
117         this.cachePolicy == Alloy.ViewModel.CachePolicy.Disabled ||
118         // cache policy is to use cache only when offline and we are online
119         (this.cachePolicy == Alloy.ViewModel.CachePolicy.OfflineOnly && Titanium.Network.online)) {
120       params.skip_cache = true;
121       params.mustFetchFromWebService = true;
122     }
123 
124     // Fetch from cache if enabled
125     if (!params.skip_cache) {
126       var cachedData = this.readCachedData(url);
127       if (cachedData) {
128         this.dataReady(cachedData, params);
129       }
130       else {
131         params.mustFetchFromWebService = true;
132       }
133       
134       // Fetch from web service --
135       if (
136         // we had a cache miss, OR
137         !cachedData ||
138         // unless we want to cache once and then never call server again
139         this.cachePolicy != Alloy.ViewModel.CachePolicy.CacheAndServe) {
140         params.skip_cache = true;
141         this.fetch(params);
142       }
143       return;      
144     }
145 
146     if (params.mustFetchFromWebService && !Titanium.Network.online) {
147       var a = Titanium.UI.createAlertDialog({
148         title:'Offline',
149         message:'Sorry, you must be online for that.'
150       });
151       a.show();
152       return;
153     }
154     
155     var xhr = Titanium.Network.createHTTPClient();
156     this.currentRequest = xhr;
157     
158     xhr.open(this.httpVerb, url);
159     
160     this.setRequestHeaders(xhr, params);
161     
162     var _this = this;
163     
164     xhr.onload = function() {
165       Alloy.analytics.trackEvent('alloy:view_model:' + _this.name, 'success', url);
166 
167       if (!params.async)
168         Ti.App.fireEvent('app:hide.loader');
169 
170       //if(this.status >= 400) {
171       //  this.onerror();
172       //  return;
173       //}
174       _this.currentRequest = null;
175       
176       //info("xhr Call returned");
177       //info(this.responseText);
178       var data;
179       try {
180         data = _this.parseData(this);
181       }
182       catch(e) {}
183       
184       if (!data) {
185         _this.onError(this, params);
186         return;
187       }
188       
189       if (data && !_this.blocked) {
190         _this.dataReady(data, params);
191         if (_this.view && _this.view.dataReady)
192           _this.view.dataReady(data, params, _this);
193           
194         if (_this.cachePolicy != Alloy.ViewModel.CachePolicy.Disabled) {
195           _this.cacheData(url, data);
196         }
197       }
198     }
199     
200     xhr.onerror = function() {
201       Alloy.analytics.trackEvent('alloy:view_model:' + _this.name, 'error', url);
202       _this.onError(this, params);
203       if (!params.async)
204         Ti.App.fireEvent('app:hide.loader');
205       _this.currentRequest = null;
206     }
207     
208     if (!params.async)
209       Ti.App.fireEvent('app:show.loader');
210     
211     this.sendRequest(xhr, params);
212   },
213 
214   sendRequest: function(xhr, params) {
215     xhr.send();
216   },
217   
218   parseData: function(response) {
219     return eval('('+response.responseText+')');
220   },
221   
222   readCachedData: function(url) {
223     //info("Reading cached data for url " + url);
224     var data = null;
225     
226     try {
227       var sql = 'SELECT data from view_model_cache WHERE url = ?;';
228       var rows = Alloy.Database.getInstance().db.execute(sql, this.getCacheKey(url));
229       if (rows.getRowCount() > 0 && rows.isValidRow()) {
230         data = eval('(' + rows.fieldByName('data') + ')');
231       }
232       rows.close();
233     }
234     catch(exception) {
235       Ti.API.error(exception);
236       return null;
237     }
238     //info("Data returned: " + JSON.stringify(data));
239     return data;
240   },
241   
242   cacheData: function(url, data) {
243     //info("Caching data for url: " + url);
244     try {
245       var sql = 'INSERT OR REPLACE INTO view_model_cache (url,data) VALUES (?,?)';
246       Alloy.Database.getInstance().db.execute(sql, this.getCacheKey(url), JSON.stringify(data));
247     }
248     catch(exception) {
249       Ti.API.error(exception);
250     }
251     //info("Done");
252   },
253   
254   getCacheKey: function(url) {
255     return url;
256   },
257   
258   dataReady: function(data, params) {
259     this.data = data;
260   },
261   
262   onError: function(response) {
263     if (response.status == 0) {
264       Ti.App.fireEvent('app:network:timeout');
265     }
266     
267     if (this.view && this.view.onError)
268       this.view.onError(response, this);
269 
270     if (this.showLoginViewOn403 && response.status == 403) {
271       Ti.App.fireEvent('app:login:show');
272     }
273     
274     // handle network error
275     warn(JSON.stringify(response));
276   },

278   sameObjects: function(obj1, obj2) {
279     if (!obj1.json) obj1.json = JSON.stringify(obj1);
280     if (!obj2.json) obj2.json = JSON.stringify(obj2);
281     
282     if (!obj1.md5) obj1.md5 = md5(obj1.json);
283     if (!obj2.md5) obj2.md5 = md5(obj2.json);
284     
285     var result = (obj1.md5 == obj2.md5) && (obj1.json == obj2.json);
286     if (!result) {
287       //info("    obj1: " + obj1.md5 + " -- " + obj1.json);
288       //info("    obj2: " + obj2.md5 + " -- " + obj2.json);
289     }
290     return result;
291   },
